document.addEventListener("DOMContentLoaded", function() {
    function animateTitle(selector, direction) {
        const element = document.querySelector(selector);
        if (element) {
            element.style.visibility = 'visible'; // Hacer visible el título justo antes de animarlo
            const text = element.innerHTML; // Usar innerHTML para conservar el formato
            element.innerHTML = ''; // Clear the element's inner text

            // Crear un elemento temporal para procesar el HTML
            const tempElement = document.createElement('div');
            tempElement.innerHTML = text;

            // Crear un fragmento de documento
            const fragment = document.createDocumentFragment();

            // Iterar sobre los nodos hijos del elemento temporal
            tempElement.childNodes.forEach(node => {
                if (node.nodeType === Node.TEXT_NODE) {
                    // Si es un nodo de texto, animar las letras
                    const words = node.textContent.split(' ');
                    words.forEach((word, wordIndex) => {
                        const wordSpan = document.createElement('span');
                        wordSpan.classList.add('word');
                        fragment.appendChild(wordSpan); // Añadir al fragmento
                        const letters = word.split('');
                        letters.forEach((letter, letterIndex) => {
                            const span = document.createElement('span');
                            span.classList.add('letter');
                            const animationName = direction === 'left' ? 'flyInFromLeft' : 'flyInFromRight';
                            span.textContent = letter;
                            span.style.animation = `${animationName} 1s ease-out forwards ${(wordIndex * 0.5) + (letterIndex * 0.1)}s`;
                            wordSpan.appendChild(span);
                        });
                        const spaceSpan = document.createElement('span');
                        spaceSpan.textContent = ' ';
                        fragment.appendChild(spaceSpan); // Añadir al fragmento
                    });
                } else {
                    // Si es un nodo HTML, agregarlo al fragmento
                    fragment.appendChild(node.cloneNode(true));
                }
            });

            // Añadir el fragmento al elemento principal
            element.appendChild(fragment);
        }
    }

    // Animar el título principal desde la izquierda
    animateTitle("#titulo-principal", 'left');

    // Animar el título web desde la derecha
    animateTitle("#titulo-web", 'right');

        const elemento = document.getElementById("mecanografiado");
        let i = 0;
        const velocidad = 50;
        let htmlTemp = '';
        let etiquetaAbierta = false; // Variable para controlar si hay una etiqueta abierta
      
        function mecanografiar() {
          if (i < texto.length) {
            if (texto[i] === '<') {
              etiquetaAbierta = true; // Se abre una etiqueta
            } else if (texto[i] === '>') {
              etiquetaAbierta = false; // Se cierra una etiqueta
              elemento.innerHTML = htmlTemp; // Actualizar el innerHTML solo cuando la etiqueta se cierra
            }
      
            htmlTemp += texto[i]; // Añadir el carácter actual al HTML temporal
            
            if (!etiquetaAbierta) { 
              // Si no hay una etiqueta abierta, mostrar el texto
              elemento.innerHTML = htmlTemp;
            }
      
            i++;
            setTimeout(mecanografiar, velocidad);
          }
        }
